% 定数を操作
%% 変数1に定数10を代入
<Set: 1, 10>
<SET: 1, 10>
<set: 1, 10>
<代入: 1, 10>
<=: 1, 10>

%% 変数2に定数20を加算
<Add: 2, 20>
<ADD: 2, 20>
<add: 2, 20>
<加算: 2, 20>
<+: 2, 20>

%% 変数3に定数30を減算
<Sub: 3, 30>
<SUB: 3, 30>
<sub: 3, 30>
<減算: 3, 30>
<-: 3, 30>

%% 変数4に定数40を乗算
<Mul: 4, 40>
<MUL: 4, 40>
<mul: 4, 40>
<乗算: 4, 40>
<*: 3, 30>

%% 変数5に定数50を除算
<Div: 5, 50>
<DIV: 5, 50>
<div: 5, 50>
<除算, 5, 50>
</: 3, 30>

%% 変数6に定数60の剰余
<Mod: 6, 60>
<MOD: 6, 60>
<mod: 6, 60>
<剰余: 6, 60>
<%: 6, 60>

%% 変数1~30に定数10を代入
<Set: 1-30, 10>
<SET: 1-30, 10>
<set: 1-30, 10>
<代入: 1-30, 10>
<=: 1-30, 10>

%% 変数2~40に定数20を加算
<Add: 2-40, 20>
<ADD: 2-40, 20>
<add: 2-40, 20>
<加算: 2-40, 20>
<+: 2-40, 20>

%% 変数3~50に定数30を減算
<Sub: 3-50, 30>
<SUB: 3-50, 30>
<sub: 3-50, 30>
<減算: 3-50, 30>
<-: 3-50, 30>

%% 変数4~60に定数40を乗算
<Mul: 4-60, 40>
<MUL: 4-60, 40>
<mul: 4-60, 40>
<乗算: 4-60, 40>
<*: 4-60, 40>

%% 変数5~70に定数50を除算
<Div: 5-70, 50>
<DIV: 5-70, 50>
<div: 5-70, 50>
<除算, 5-70, 50>
</, 5-70, 50>

%% 変数6~80に定数6の剰余を代入
<Mod: 6-80, 60>
<MOD: 6-80, 60>
<mod: 6-80, 60>
<剰余: 6-80, 60>
<%: 6-80, 60>
